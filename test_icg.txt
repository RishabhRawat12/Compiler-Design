void calculate() {
    int score = 85;
    int bonus = 10;
    int total = score + bonus * 2;
    
    if (total > 100) {
        total = 100;
    } else {
        total = total + 5;
    }

    for (int i = 0; i < 5; i++) {
        total = total + i;
    }
}

int main() {
    calculate();
    return 0;
}
